# OpenAI API中转调用示例 (Python)
import requests
import json

API_KEY = "your_4z_api_key"
API_URL = "https://api.zzzzapi.com/v1/chat/completions"  # OpenAI API中转地址

headers = {
    "Authorization": f"Bearer {API_KEY}",
    "Content-Type": "application/json"

    "model": "gpt-4o",  # 支持所有OpenAI模型
    "messages": [
        {"role": "system", "content": "你是一个专业的AI助手"},
        {"role": "user", "content": "请解释OpenAI API的主要功能和应用场景"}
    ],
    "temperature": 0.7,
    "max_tokens": 2000
}

response = requests.post(API_URL, headers=headers, json=data)
result = response.json()
print(json.dumps(result, indent=4, ensure_ascii=False))
        

# NodeJS使用OpenAI API中转服务示例
const axios = require('axios');

const API_KEY = 'your_4z_api_key';
const API_URL = 'https://api.zzzzapi.com/v1/chat/completions';

async function callOpenAIAPI() {
  try {
    const response = await axios.post(
      API_URL,
      {
        model: 'gpt-4',
        messages: [
          { role: 'system', content: '你是一个专业的AI助手' },
          { role: 'user', content: '请解释OpenAI API的主要功能和应用场景' }
        ],
        temperature: 0.7,
        max_tokens: 2000
      },
      {
        headers: {
          'Authorization': `Bearer ${API_KEY}`,
          'Content-Type': 'application/json'
        }
      }
    );
    
    console.log(JSON.stringify(response.data, null, 2));
  } catch (error) {
    console.error('API调用失败:', error.response?.data || error.message);
  }
}

callOpenAIAPI();
